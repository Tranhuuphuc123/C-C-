>>>>> TRUYỀN THAM SỐ MẢNG CHO FUNCTION:
  - Syntax:
      xd funtion trc rồi truyền tham số mảng arr vào kèm các đối số như mảng 1 chiều:
       + lưu ý khi truyền mảng 2 chiều vào function:
         ++ Có thể bỏ phần tử khai báo dòng và chỉ cần khai báo phần tử số cột thui
         ++ Hoặc có thể truyền con trỏ và phần tử số cột
      
       <tyep><function_name>(int [*arr][n],int &m, int &n ){}; 
          or
       <tyep><function_name>(int arr[][n],int &m, int &n ){};

       ===> lưu ý truyền đối số vào mảng có thể int m, int n hoặc int &m, int &n.

      using namespace std;

      // tạo biến hắng số sl tối đa row and col của mảng có thể đạt đc
      #define ROW 100
      #define COL 100

      /* xây dựng mảng nhập 2 chiều: int &r, int &c là cách truyền số
       phần tử cột và dòng được hiên thị chỉ định từ người dùng cho mảng.. còn COL, ROW 100 là giới hạn có thể đạt đc thui
      */
      void Nhapmang(int arr[][COL],int &r, int &c){
         cout<<"nhap gioi han so phan tu row: "<<endl;
         cin>>r;
         cout<<"nhap gioi han so phan tu col: "<<endl;
         cin>>c;

         for(int i = 0;i<r;i++){
           for(int j = 0; j<c;j++){
               cout<<"arr["<<i<<"]"<<"["<<j<<"]: "<<endl;
               cin>>arr[i][j];
           }
         }
      }


      // function xuất mảng
